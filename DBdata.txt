CREATE DATABASE tryeat;

USE tryeat;

CREATE TABLE user(
	user_id INT(11) NOT NULL AUTO_INCREMENT,
	user_login_id VARCHAR(30) NOT NULL,
	user_pwd VARCHAR(50) NOT NULL,
	profile VARCHAR(100) NULL,
	signdate DATETIME NOT NULL,
	PRIMARY KEY (user_id)
);

CREATE TABLE images (
	img_id INT(10) NOT NULL auto_increment,
	restaurant_id INT(11) NULL,
	user_id INT(11) NULL,
	image MEDIUMBLOB NOT NULL,
	PRIMARY KEY (img_id),
	FOREIGN KEY (user_id) REFERENCES user (user_id)
);

CREATE TABLE follow(
	user_id INT(11) NULL,
	follow_id INT(11) NOT NULL,
	target_id VARCHAR(100) NOT NULL,
	fdate DATETIME NOT NULL,
	FOREIGN KEY (user_id) REFERENCES user (user_id)
);

CREATE TABLE restaurant(
	restaurant_id INT(11) NOT NULL AUTO_INCREMENT,
	img_id INT(10) NULL,
	restaurant_name VARCHAR(30) NOT NULL,
	locate_latitude VARCHAR(100) NOT NULL,
	locate_longitude VARCHAR(100) NOT NULL,
	restaurant_desc VARCHAR(100) NULL,
	open_time TIME NOT NULL,
	close_time TIME NOT NULL,
	review_count INT(11) NULL,
	average INT(11) NULL,
	PRIMARY KEY (restaurant_id),
	FOREIGN KEY (img_id) REFERENCES images (img_id)
);

CREATE TABLE rst_rate(
	txt_id INT(11) NOT NULL AUTO_INCREMENT,
	restaurant_id INT(11) NULL,
	user_id INT(11) NULL,
	img_id INT(10) NULL,
	title VARCHAR(100) NOT NULL,
	content TEXT NOT NULL,
	date DATETIME NOT NULL,
	rate INT(11) NOT NULL,
	PRIMARY KEY (txt_id),
	FOREIGN KEY (restaurant_id) REFERENCES restaurant (restaurant_id),
	FOREIGN KEY (user_id) REFERENCES user (user_id),
	FOREIGN KEY (img_id) REFERENCES images (img_id)
);

CREATE TABLE ad(
	ad_id INT(11) NOT NULL AUTO_INCREMENT,
	restaurant_id INT(11) NULL,
	ad_requester VARCHAR(100) NOT NULL,
	start_date DATETIME NOT NULL,
	ad_period INT(11) NOT NULL,
	PRIMARY KEY (ad_id),
	FOREIGN KEY (restaurant_id) REFERENCES restaurant (restaurant_id)
);
